/* ── ESTADO GLOBAL ─────────────────────────────────────────────────── */
static float             g_pa = 0.0f, g_tc = 0.0f;
static SemaphoreHandle_t g_data_mux;

static int               g_ws_fds[MAX_WS];
static SemaphoreHandle_t g_ws_mux;
static httpd_handle_t    g_srv = NULL;

static i2c_master_bus_handle_t g_i2c_bus = NULL;
static i2c_master_dev_handle_t g_i2c_smp = NULL;

/* ══════════════════════════════════════════════════════════════════════
   SENSOR SMP3011
   ══════════════════════════════════════════════════════════════════════ */
static void smp3011_init(void)
{
    i2c_master_bus_config_t bus_cfg = {
        .clk_source            = I2C_CLK_SRC_DEFAULT,
        .glitch_ignore_cnt     = 7,
        .i2c_port              = I2C_BUS_PORT,
        .sda_io_num            = I2C_SDA,
        .scl_io_num            = I2C_SCL,
        .flags.enable_internal_pullup = true,
    };
    ESP_ERROR_CHECK(i2c_new_master_bus(&bus_cfg, &g_i2c_bus));

    i2c_device_config_t dev_cfg = {
        .dev_addr_length = I2C_ADDR_BIT_LEN_7,
        .device_address  = SMP3011_ADDR,
        .scl_speed_hz    = 10000,    /* 10kHz: máxima compatibilidade com pull-ups internos fracos */
        .scl_wait_us     = 1000000, /* 1s timeout para clock stretching do SMP3011 */
    };
    ESP_ERROR_CHECK(i2c_master_bus_add_device(g_i2c_bus, &dev_cfg, &g_i2c_smp));

    uint8_t cmd = 0xAC;
    i2c_master_transmit(g_i2c_smp, &cmd, 1, 100);
    vTaskDelay(pdMS_TO_TICKS(100));
    ESP_LOGI(TAG, "SMP3011 iniciado (I2C addr=0x%02X)", SMP3011_ADDR);
}

static void i2c_full_reinit(void)
{
    if (g_i2c_smp) { i2c_master_bus_rm_device(g_i2c_smp); g_i2c_smp = NULL; }
    if (g_i2c_bus) { i2c_del_master_bus(g_i2c_bus);        g_i2c_bus = NULL; }
    vTaskDelay(pdMS_TO_TICKS(200));
    gpio_reset_pin(I2C_SDA);
    gpio_reset_pin(I2C_SCL);
    vTaskDelay(pdMS_TO_TICKS(100));
    smp3011_init();
}

static void sensor_task(void *_)
{
    int fails = 0;
    while (1) {
        uint8_t buf[6];
        esp_err_t ret = i2c_master_receive(g_i2c_smp, buf, sizeof(buf), 100);

        if (ret != ESP_OK) {
            fails++;
            ESP_LOGW(TAG, "I2C erro: %s (%d)", esp_err_to_name(ret), fails);
            if (fails >= 3) {
                ESP_LOGW(TAG, "Reiniciando driver I2C completo");
                i2c_full_reinit();
                fails = 0;
            }
            vTaskDelay(pdMS_TO_TICKS(300));
            continue;
        }
        fails = 0;

        if ((buf[0] & 0x20) == 0) {
            /* Dispara próxima medição */
            uint8_t cmd = 0xAC;
            i2c_master_transmit(g_i2c_smp, &cmd, 1, 100);

            /* Converte pressão: saída de 24 bits → Pa */
            float pa = (float)(((uint32_t)buf[1] << 16) |
                                ((uint32_t)buf[2] <<  8) |
                                 (uint32_t)buf[3]);
            pa = (pa / 16777215.0f - 0.15f) / 0.7f * 500000.0f;

            /* Converte temperatura: saída de 16 bits → °C */
            float tc = (float)(((uint32_t)buf[4] << 8) | buf[5]);
            tc = (190.0f * tc / 65535.0f) - 40.0f;

            if (xSemaphoreTake(g_data_mux, pdMS_TO_TICKS(10)) == pdTRUE) {
                g_pa = pa;
                g_tc = tc;
                xSemaphoreGive(g_data_mux);
            }
        } else {
            ESP_LOGD(TAG, "Sensor ocupado (status=0x%02X)", buf[0]);
        }

        vTaskDelay(pdMS_TO_TICKS(100));
    }
}

/* ══════════════════════════════════════════════════════════════════════
   WEBSOCKET
   ══════════════════════════════════════════════════════════════════════ */
static esp_err_t ws_handler(httpd_req_t *req)
{
    if (req->method == HTTP_GET) {
        /* Novo cliente — registra o file descriptor */
        int fd = httpd_req_to_sockfd(req);
        if (xSemaphoreTake(g_ws_mux, pdMS_TO_TICKS(200)) == pdTRUE) {
            for (int i = 0; i < MAX_WS; i++) {
                if (g_ws_fds[i] < 0) {
                    g_ws_fds[i] = fd;
                    ESP_LOGI(TAG, "WS conectado: fd=%d slot=%d", fd, i);
                    break;
                }
            }
            xSemaphoreGive(g_ws_mux);
        }
        return ESP_OK;
    }

    /* Absorve frames recebidos (ping/close do browser) */
    uint8_t tmp[64] = {0};
    httpd_ws_frame_t pkt = { .payload = tmp };
    return httpd_ws_recv_frame(req, &pkt, sizeof(tmp) - 1);
}

static void ws_push_task(void *_)
{
    while (1) {
        vTaskDelay(pdMS_TO_TICKS(WS_INTERVAL_MS));
        if (!g_srv) continue;

        float pa, tc;
        if (xSemaphoreTake(g_data_mux, pdMS_TO_TICKS(10)) != pdTRUE) continue;
        pa = g_pa;
        tc = g_tc;
        xSemaphoreGive(g_data_mux);

        char json[128];
        int  len = snprintf(json, sizeof(json),
                            "{\"pressure_pa\":%.2f,\"temperature_c\":%.2f}", pa, tc);

        httpd_ws_frame_t pkt = {
            .final     = true,
            .fragmented = false,
            .type      = HTTPD_WS_TYPE_TEXT,
            .payload   = (uint8_t *)json,
            .len       = (size_t)len,
        };

        if (xSemaphoreTake(g_ws_mux, pdMS_TO_TICKS(10)) != pdTRUE) continue;
        for (int i = 0; i < MAX_WS; i++) {
            if (g_ws_fds[i] >= 0) {
                if (httpd_ws_send_frame_async(g_srv, g_ws_fds[i], &pkt) != ESP_OK) {
                    ESP_LOGI(TAG, "WS fd=%d desconectado", g_ws_fds[i]);
                    g_ws_fds[i] = -1;
                }
            }
        }
        xSemaphoreGive(g_ws_mux);
    }
}

/* ══════════════════════════════════════════════════════════════════════
   HTTP HANDLERS
   ══════════════════════════════════════════════════════════════════════ */
static esp_err_t sobre_handler(httpd_req_t *req)
{
    httpd_resp_set_type(req, "text/html; charset=UTF-8");
    return httpd_resp_send(req, sobre_html_start, sobre_html_end - sobre_html_start);
}

static esp_err_t medidor_handler(httpd_req_t *req)
{
    httpd_resp_set_type(req, "text/html; charset=UTF-8");
    return httpd_resp_send(req, index_html_start, index_html_end - index_html_start);
}

static esp_err_t codigo_handler(httpd_req_t *req)
{
    static const char hdr[] =
        "<!DOCTYPE html><html lang='pt-br'><head><meta charset='UTF-8'>"
        "<title>C\xC3\xB3digo \xE2\x80\x93 Aferidor</title>"
        "<style>*{box-sizing:border-box;margin:0;padding:0;}"
        "body{background:#1e1e2e;color:#cdd6f4;font-family:Arial,sans-serif;}"
        "header{background:#041e64;color:#fff;padding:28px 20px 16px;text-align:center;font-style:italic;}"
        "header h1{font-size:1.8rem;letter-spacing:2px;margin-bottom:10px;}"
        ".menu{display:flex;justify-content:center;gap:30px;font-size:17px;}"
        ".menu a{color:#fff;text-decoration:none;}.menu a:hover{color:#ffd700;}"
        "pre{padding:30px;font-size:14px;line-height:1.7;white-space:pre-wrap;word-break:break-word;}"
        "</style></head><body>"
        "<header><h1>C\xC3\xB3DIGO FONTE \xE2\x80\x93 main.c</h1>"
        "<nav><div class='menu'>"
        "<a href='/'>Sobre</a>"
        "<a href='/codigo'>C\xC3\xB3digo</a>"
        "<a href='/medidor'>Medidor ao Vivo</a>"
        "</div></nav></header><pre>";
    static const char ftr[] = "</pre></body></html>";

    httpd_resp_set_type(req, "text/html; charset=UTF-8");
    httpd_resp_send_chunk(req, hdr, HTTPD_RESP_USE_STRLEN);

    /* Envia main.c escapando caracteres HTML */
    const char *p = main_src_txt_start, *end = main_src_txt_end;
    char buf[512]; int bi = 0;
    while (p < end) {
        char c = *p++;
        const char *esc = (c=='<') ? "&lt;" : (c=='>') ? "&gt;" : (c=='&') ? "&amp;" : NULL;
        if (esc) {
            if (bi) { httpd_resp_send_chunk(req, buf, bi); bi = 0; }
            httpd_resp_send_chunk(req, esc, HTTPD_RESP_USE_STRLEN);
        } else {
            buf[bi++] = c;
            if (bi == (int)sizeof(buf)) { httpd_resp_send_chunk(req, buf, bi); bi = 0; }
        }
    }
    if (bi) httpd_resp_send_chunk(req, buf, bi);
    httpd_resp_send_chunk(req, ftr, HTTPD_RESP_USE_STRLEN);
    httpd_resp_send_chunk(req, NULL, 0);
    return ESP_OK;
}

#define IMG_HANDLER(name, sym, mime) \
static esp_err_t name(httpd_req_t *req){ \
    httpd_resp_set_type(req, mime); \
    return httpd_resp_send(req,(const char*)sym##_start,(ssize_t)(sym##_end-sym##_start)); \
}

/* ══════════════════════════════════════════════════════════════════════
   HTTP SERVER
   ══════════════════════════════════════════════════════════════════════ */
static void start_server(void)
{
    httpd_config_t cfg      = HTTPD_DEFAULT_CONFIG();
    cfg.max_open_sockets    = 7;
    cfg.max_uri_handlers    = 12;
    cfg.lru_purge_enable    = true;

    ESP_ERROR_CHECK(httpd_start(&g_srv, &cfg));

    static const httpd_uri_t uri_index = {
        .uri     = "/",
        .method  = HTTP_GET,
        .handler = sobre_handler,
    };
    static const httpd_uri_t uri_sobre = {
        .uri     = "/sobre",
        .method  = HTTP_GET,
        .handler = sobre_handler,
    };
    static const httpd_uri_t uri_medidor = {
        .uri     = "/medidor",
        .method  = HTTP_GET,
        .handler = medidor_handler,
    };
    static const httpd_uri_t uri_codigo = {
        .uri     = "/codigo",
        .method  = HTTP_GET,
        .handler = codigo_handler,
    };
    static const httpd_uri_t uri_ws = {
        .uri          = "/ws",
        .method       = HTTP_GET,
        .handler      = ws_handler,
        .is_websocket = true,
    };

    httpd_register_uri_handler(g_srv, &uri_index);
    httpd_register_uri_handler(g_srv, &uri_sobre);
    httpd_register_uri_handler(g_srv, &uri_medidor);
    httpd_register_uri_handler(g_srv, &uri_codigo);
    httpd_register_uri_handler(g_srv, &uri_ws);

#define REG_IMG(path, fn) do { \
    static const httpd_uri_t u = {.uri=path,.method=HTTP_GET,.handler=fn}; \
    httpd_register_uri_handler(g_srv, &u); \
} while(0)
    REG_IMG("/card1.jpg",      card1_handler);
    REG_IMG("/card2.jpg",      card2_handler);
    REG_IMG("/card3.jpg",      card3_handler);
    REG_IMG("/analogico.jpg",  analogico_handler);
    REG_IMG("/digital.png",    digital_handler);
    REG_IMG("/eletronico.jpg", eletronico_handler);

    ESP_LOGI(TAG, "HTTP server iniciado");
}

/* ══════════════════════════════════════════════════════════════════════
   WIFI SOFTAP
   ══════════════════════════════════════════════════════════════════════ */
static void wifi_init_ap(void)
{
    ESP_ERROR_CHECK(esp_netif_init());
    ESP_ERROR_CHECK(esp_event_loop_create_default());
    esp_netif_create_default_wifi_ap();

    wifi_init_config_t init = WIFI_INIT_CONFIG_DEFAULT();
    ESP_ERROR_CHECK(esp_wifi_init(&init));

    wifi_config_t wcfg = {
        .ap = {
            .ssid           = WIFI_SSID,
            .ssid_len       = strlen(WIFI_SSID),
            .channel        = WIFI_CHANNEL,
            .password       = WIFI_PASS,
            .max_connection = MAX_STA,
            .authmode       = WIFI_AUTH_WPA_WPA2_PSK,
        },
    };

    ESP_ERROR_CHECK(esp_wifi_set_mode(WIFI_MODE_AP));
    ESP_ERROR_CHECK(esp_wifi_set_config(WIFI_IF_AP, &wcfg));
    ESP_ERROR_CHECK(esp_wifi_start());

    ESP_LOGI(TAG, "SoftAP '%s' ativo — acesse http://192.168.4.1", WIFI_SSID);
}

/* ══════════════════════════════════════════════════════════════════════
   APP MAIN
   ══════════════════════════════════════════════════════════════════════ */
void app_main(void)
{
    /* NVS obrigatório para WiFi */
    esp_err_t r = nvs_flash_init();
    if (r == ESP_ERR_NVS_NO_FREE_PAGES || r == ESP_ERR_NVS_NEW_VERSION_FOUND) {
        ESP_ERROR_CHECK(nvs_flash_erase());
        r = nvs_flash_init();
    }
    ESP_ERROR_CHECK(r);

    /* Mutexes e lista de clientes WS */
    g_data_mux = xSemaphoreCreateMutex();
    g_ws_mux   = xSemaphoreCreateMutex();
    for (int i = 0; i < MAX_WS; i++) g_ws_fds[i] = -1;

    smp3011_init();
    wifi_init_ap();
    start_server();

    xTaskCreate(sensor_task,  "sensor",  4096, NULL, 5, NULL);
    xTaskCreate(ws_push_task, "ws_push", 4096, NULL, 4, NULL);
}
